# 函数节流和防抖

## 函数防抖
先判断后执行
```javascript
function debounce(fn, wait = 500) {
  let timer;

  return function (...args) {
    clearTimeout(timer)

    timer = setTimeout(() => {
      fn.apply(this, args)
    }, wait)
  }
}
```
## 函数节流
先执行后判断
```javascript
 let throttle = function (fn, interval) {
    let timer, firstTime = true;

    return function () {
      let args = arguments;
      if (firstTime) {  // 第一次执行不需要节流
      	fn.apply(this, args)
      	firstTime = false;
      }

      if (timer) {
      	return false;
      }

      timer = setTimeout(() => {
      	clearTimeout(timer);
      	fn.apply(this, args);
        timer = null;
      }, interval || 1000)
    }
  }
```
